// js/pi-utils.js // ============================================ // UTILITAIRES POUR CRYPTO PI // ============================================ // Taux de conversion (à mettre à jour régulièrement) let EXCHANGE_RATE = 1000; // 1 Pi = 1000 FCFA // Configurer le taux de change export function setExchangeRate(rate) { if (rate > 0) { EXCHANGE_RATE = rate; } return EXCHANGE_RATE; } // Obtenir le taux de change export function getExchangeRate() { return EXCHANGE_RATE; } // Convertir FCFA en Pi export function toPi(amountFCFA) { return amountFCFA / EXCHANGE_RATE; } // Convertir Pi en FCFA export function toFCFA(amountPi) { return amountPi * EXCHANGE_RATE; } // Formater un montant en Pi export function formatPi(amountPi) { return `${amountPi.toFixed(2)} Pi`; } // Formater un montant en FCFA export function formatFCFA(amountFCFA) { return new Intl.NumberFormat('fr-BJ', { style: 'currency', currency: 'XOF', minimumFractionDigits: 0, maximumFractionDigits: 0 }).format(amountFCFA); } // Générer le message WhatsApp pour un paiement Pi export function generatePiPaymentMessage(productData, sellerData) { const priceInPi = toPi(productData.price).toFixed(2); return `Bonjour 👋 Je souhaite acheter votre produit : "${productData.title}" sur Bénin Pi Market. 💰 Prix en Pi : ${priceInPi} Pi (soit ${formatFCFA(productData.price)}) Je suis prêt(e) à effectuer le paiement en Crypto Pi. Pouvez-vous me confirmer votre adresse Pi pour le paiement ? Merci 🙏`; } // Valider un portefeuille Pi (format simple) export function isValidPiWallet(address) { // Format Pi Wallet: G + 25 caractères alphanumériques (exemple) return /^G[A-Z0-9]{25}$/.test(address.toUpperCase()); } // Obtenir le lien WhatsApp pour paiement Pi export function getPiWhatsAppLink(sellerWhatsapp, productData) { const message = generatePiPaymentMessage(productData); return `https://wa.me/${sellerWhatsapp.replace(/\s/g, '').replace(/^\+/, '')}?text=${encodeURIComponent(message)}`; } // Prix Pi suggéré (arrondi) export function suggestPiPrice(priceFCFA) { const piPrice = toPi(priceFCFA); return Math.round(piPrice * 10) / 10; }
